# histogram plotting function
def print_hist(col_name):
    df[col_name].plot.hist()
    plt.title(col_name)
    plt.show()
    plt.clf()

# plot all columns in dataframe
for col_name in df.columns:
    print_hist(col_name)

def print_dist(col_name):
